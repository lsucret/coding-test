class Solution {
    public int solution(int n, int w, int num) {
        
        int row = findRow(w, num);
        int column = findColumn(w, num, row);
        System.out.println("row:"+row);
        System.out.println("column:"+column);
        
        int temp = num;
        int tempRow = row;
        int answer = 0;
        while (temp <= n) {
            answer++;
            if (tempRow % 2 == 0) { // 0, 2, ...
                System.out.println("even");
                temp += (w - column - 1) * 2 + 1;
            } else {
                System.out.println("odd");
                temp += column * 2 + 1;//+1은 위쪽 이동
            }
            tempRow++;
            System.out.println("temp:"+temp);
        }
        
        return answer;
    }
    private int findRow(int w, int num) { // return : 바닥에서부터 0, 1, 2, ...
        if (num == 1) {
            return 0;
        }
        return (num - 1) / w;
    }
    private int findColumn(int w, int num, int row) { // 0, 1, 2, ..., w-1
        if (row % 2 == 0) { // 0, 2, 4, ...
            return (num - row * w) - 1;
        } else { // 1, 3, 5, ...
            return ( (row + 1) * w - num);
        }
    }
    
}
